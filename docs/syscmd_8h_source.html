123 CmdResult syscmd(const std::string &exe,
124 const std::vector<std::string> &args,
125 const std::map<std::string, std::string> &kwargs)
126 {
127 std::string cmd = exe;
128 for (const std::string &arg : args)
129 cmd += " " + arg;
130 for (const auto &kwarg : kwargs)
131 cmd += " " + kwarg.first + " " + kwarg.second;
132 cmd += " 2>&1";
133 FILE *pipe = POPEN(cmd.c_str(), "r");
134 if (!pipe)
135 {
136 CmdResult cr;
137 cr.status_ = -1;
138 cr.ptr_ = nullptr;
139 return cr;
140 }
141 std::array<char, 256> buffer;
142 std::unique_ptr<std::string> output(new std::string);
143 while (fgets(buffer.data(), buffer.size(), pipe) != nullptr)
144 *output += buffer.data();
145 int status = PCLOSE(pipe);
146 CmdResult cr;
147 cr.status_ = status;
148 cr.ptr_ = std::move(output);
149 return cr;
150 }
151}
