211 void write_lines(const std::vector<std::string> &lines) const
212 {
213 std::ofstream fout(path, std::ios::binary);
214 if (lines.empty())
215 return;
216 if (!fout.is_open())
217 throw file_error("Cannot Open File \"" + path + '"');
218 auto it = lines.begin();
219 fout << *it;
220 while (++it != lines.end())
221 fout << '\n'
222 << *it;
223 if (!fout.good())
224 throw file_error("The Stream of \"" + path + "\" is Not Good");
225 }
