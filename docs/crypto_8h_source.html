148 void transform(uint32_t *state, const uint8_t *data)
149 {
150 uint32_t W[64];
151 uint32_t A, B, C, D, E, F, G, H, T1, T2;
152 for (int i = 0; i < 16; i++)
153 W[i] = (data[i * 4] << 24) | (data[i * 4 + 1] << 16) |
154 (data[i * 4 + 2] << 8) | data[i * 4 + 3];
155 for (int i = 16; i < 64; i++)
156 W[i] = gamma1(W[i - 2]) + W[i - 7] + gamma0(W[i - 15]) + W[i - 16];
157 A = state[0];
158 B = state[1];
159 C = state[2];
160 D = state[3];
161 E = state[4];
162 F = state[5];
163 G = state[6];
164 H = state[7];
165 for (int i = 0; i < 64; i++)
166 {
167 T1 = H + sigma1(E) + ch(E, F, G) + K[i] + W[i];
168 T2 = sigma0(A) + maj(A, B, C);
169 H = G;
170 G = F;
171 F = E;
172 E = D + T1;
173 D = C;
174 C = B;
175 B = A;
176 A = T1 + T2;
177 }
178 state[0] += A;
179 state[1] += B;
180 state[2] += C;
181 state[3] += D;
182 state[4] += E;
183 state[5] += F;
184 state[6] += G;
185 state[7] += H;
186 }

334 void transform(uint32_t *state, const uint8_t *block)
335 {
336 uint32_t a = state[0], b = state[1], c = state[2], d = state[3];
337 uint32_t M[16];
338 for (int i = 0; i < 16; i++)
339 {
340 M[i] = block[i * 4] | (block[i * 4 + 1] << 8) |
341 (block[i * 4 + 2] << 16) | (block[i * 4 + 3] << 24);
342 }
343 for (int i = 0; i < 64; i++)
344 {
345 uint32_t f, g;
346 if (i < 16)
347 {
348 f = F(b, c, d);
349 g = i;
350 }
351 else if (i < 32)
352 {
353 f = G(b, c, d);
354 g = (5 * i + 1) % 16;
355 }
356 else if (i < 48)
357 {
358 f = H(b, c, d);
359 g = (3 * i + 5) % 16;
360 }
361 else
362 {
363 f = I(b, c, d);
364 g = (7 * i) % 16;
365 }
366 uint32_t temp = d;
367 d = c;
368 c = b;
369 b = b + rotate_left(a + f + K[i] + M[g], S[i]);
370 a = temp;
371 }
372 state[0] += a;
373 state[1] += b;
374 state[2] += c;
375 state[3] += d;
376 }

412 std::string base64_encode(const std::string &input)
413 {
414 const char *base64_chars = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/";
415 std::string result;
416 result.reserve(((input.size() + 2) / 3) * 4);
417 uint32_t buffer = 0;
418 int bits_collected = 0;
419 for (char c : input)
420 {
421 buffer = (buffer << 8) | static_cast<uint8_t>(c);
422 bits_collected += 8;
423 while (bits_collected >= 6)
424 {
425 bits_collected -= 6;
426 result.push_back(base64_chars[(buffer >> bits_collected) & 0x3F]);
427 }
428 }
429 if (bits_collected > 0)
430 {
431 buffer <<= (6 - bits_collected);
432 result.push_back(base64_chars[buffer & 0x3F]);
433 }
434 while (result.size() % 4)
435 result.push_back('=');
436 return result;
437 }

464 bits_collected -= 8;

616 void mix_columns(uint8_t *state)
617 {
618 uint8_t temp[4];
619 for (int i = 0; i < 4; i++)
620 {
621 int idx = i * 4;
622 temp[0] = gmul(state[idx], 2) ^ gmul(state[idx + 1], 3) ^
623 state[idx + 2] ^ state[idx + 3];
624 temp[1] = state[idx] ^ gmul(state[idx + 1], 2) ^

672 void key_expansion(const uint8_t *key, uint8_t *round_keys)
673 {
674 uint8_t temp[4];
675 uint8_t rcon[4];
676 for (int i = 0; i < 16; i++)
677 round_keys[i] = key[i];
678 for (int i = 4; i < 44; i++)
679 {
680 std::memcpy(temp, &round_keys[(i - 1) * 4], 4);
681 if (i % 4 == 0)
682 {
683 uint8_t t = temp[0];
684 temp[0] = Sbox[temp[1]];
685 temp[1] = Sbox[temp[2]];
686 temp[2] = Sbox[temp[3]];
687 temp[3] = Sbox[t];
688 rcon[0] = 0x01 << ((i / 4) - 1);
689 rcon[1] = rcon[2] = rcon[3] = 0;
690 if (rcon[0] == 0)
691 rcon[0] = 0x1b;
692 for (int j = 0; j < 4; j++)
693 temp[j] ^= rcon[j];
694 }
695 for (int j = 0; j < 4; j++)
696 {
697 round_keys[i * 4 + j] = round_keys[(i - 4) * 4 + j] ^ temp[j];
698 }
699 }
