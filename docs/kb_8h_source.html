301 int readRaw()
302 {
303#ifdef _WIN32
304 if (_kbhit())
305 return _getch();

307#else
308 struct pollfd pfd = {STDIN_FILENO, POLLIN, 0};

353 Key parse(int raw)
354 {
355 if ((raw >= 'A' && raw <= 'Z') || (raw >= '0' && raw <= '9'))
356 {
357 return static_cast<Key>(raw);
358 }
359 if (raw >= 'a' && raw <= 'z')
360 {
361 return static_cast<Key>(raw - 32);
362 }
363 switch (raw)
364 {
365 case ' ':
366 return Key::Space;
367 case 13:
368 return Key::Enter;
369 case 27:
370 return handleEscape();
371 case 8:
372 case 127:
373 return Key::Backspace;
374 case 9:
375 return Key::Tab;
376 }
377#ifdef _WIN32
378 if (raw == 0xE0 || raw == 0x00)
379 {
380 int ext = _getch();
381 switch (ext)
382 {
383 case 0x48:
384 return Key::Up;
385 case 0x50:
386 return Key::Down;
387 case 0x4B:

428 Key handleEscape()
429 {
430#ifndef _WIN32
431 struct pollfd pfd = {STDIN_FILENO, POLLIN, 0};
432 if (poll(&pfd, 1, 5) <= 0)
433 return Key::Esc;
434 char seq[3];
435 int n = read(STDIN_FILENO, seq, 3);
436 if (n >= 2 && seq[0] == '[')
437 {
438 switch (seq[1])
439 {
440 case 'A':
441 return Key::Up;
442 case 'B':
443 return Key::Down;
444 case 'C':
445 return Key::Right;
446 case 'D':
447 return Key::Left;
448 }
449 }
450#endif
451 return Key::Esc;
452 }
