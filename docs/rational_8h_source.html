65 static T gcd(T a, T b)
66 {
67 a = std::abs(a);
68 b = std::abs(b);
69 while (b != 0)
70 {
71 T temp = b;
72 b = a % b;
73 a = temp;
74 }
75 return a;
76 }

82 void reduce()
83 {
84 if (den == 0)
85 {
86 throw domain_error("Denominator cannot be zero");
87 }
88 if (den < 0)
89 {
90 num = -num;
91 den = -den;
92 }
93 T g = gcd(num, den);
94 num /= g;
95 den /= g;
96 }
