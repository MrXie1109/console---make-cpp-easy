54 template <class Cont>
55 std::ostream &cont_print_sequence(std::ostream &os, const Cont &cont)
56 {
57 if (begin(cont) == end(cont))
58 return os << "[]";
59 auto it = begin(cont);
60 os << '[';
61 repr(*it, os);
62 while (++it != end(cont))
63 {
64 os << ", ";
65 repr(*it, os);
66 }
67 return os << ']';
68 }
