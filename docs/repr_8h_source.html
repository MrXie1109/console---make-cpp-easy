56 std::string tiname(const std::type_info &ti)
57 {
58#ifdef _MSC_VER
59 return ti.name();
60#elif defined(__GNUG__)
61 int status = 0;
62 std::unique_ptr<char, void (*)(void *)> result(
63 abi::__cxa_demangle(ti.name(), nullptr, nullptr, &status),
64 std::free);
65 return (status == 0) ? result.get() : ti.name();
66#else
67 return ti.name();
68#endif
69 }
