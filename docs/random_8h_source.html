175 template <class C>
176 void shuffle(C &&c, std::mt19937 &gen = default_gen())
177 {
178 if (std::begin(c) == std::end(c))
179 return;
180 for (size_t i = c.size() - 1; i > 0; i--)
181 {
182 auto j = randint<size_t>(0, i, gen);
183 std::swap(*std::next(std::begin(c), i),
184 *std::next(std::begin(c), j));
185 }
186 }

440 template <class C>
441 std::vector<typename C::value_type>
442 sample(const C &c, size_t size, bool replace = false,
443 std::mt19937 &gen = default_gen())
444 {
445 using value_type = typename C::value_type;
446 std::vector<value_type> result;
447 result.reserve(size);
448 auto it_begin = std::begin(c);
449 auto it_end = std::end(c);
450 size_t n = std::distance(it_begin, it_end);
451 if (n == 0)
452 throw container_error(
453 "Cannot sample from empty container");
454 if (!replace && size > n)
455 throw container_error(
456 "Sample size exceeds container size when replace=false");
457 if (replace)
458 for (size_t i = 0; i < size; ++i)
459 {
460 result.push_back(*std::next(
461 it_begin,
462 std::uniform_int_distribution<size_t>(0, n - 1)(gen)));
463 }
464 else
465 {
466 std::vector<size_t> indices(n);
467 std::iota(indices.begin(), indices.end(), 0);
468 shuffle(indices, gen);
469 for (size_t i = 0; i < size; ++i)
470 result.push_back(*std::next(it_begin, indices[i]));
471 }
472 return result;
473 }
