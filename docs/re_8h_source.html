247 std::vector<std::string> split(const std::string &text,
248 int maxsplit = 0) const
249 {
250 std::vector<std::string> result;
251 std::sregex_token_iterator it(text.begin(), text.end(), pattern, -1);
252 std::sregex_token_iterator end;
253
254 int count = 0;
255 for (; it != end && (maxsplit <= 0 || count < maxsplit);
256 ++it, ++count)
257 {
258 result.push_back(*it);
259 }
260
261 if (maxsplit > 0 && count == maxsplit && it != end)
262 {
263 std::string remaining;
264 for (; it != end; ++it)
265 {
266 if (!remaining.empty())
267 remaining += *it;
268 else
269 remaining = *it;
270 }
271 result.push_back(remaining);
272 }
273
274 return result;
275 }
