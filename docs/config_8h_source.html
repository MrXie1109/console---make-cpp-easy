132 friend std::istream &operator>>(std::istream &is, INIConfig &config)
133 {
134 std::string line;
135 std::string current_section;
136 while (std::getline(is, line))
137 {
138 std::string trimmed_line = trim(line);
139 if (trimmed_line.empty() ||
140 trimmed_line[0] == ';' || trimmed_line[0] == '#')
141 continue;
142 if (trimmed_line.front() == '[' && trimmed_line.back() == ']')
143 current_section = trim(
144 trimmed_line.substr(1, trimmed_line.size() - 2));
145 else
146 {
147 auto pos = trimmed_line.find('=');
148 if (pos != std::string::npos)
149 {
150 std::string key = trim(trimmed_line.substr(0, pos));
151 std::string value = trim(trimmed_line.substr(pos + 1));
152 config.data_[current_section][key] = value;
153 }
154 }
155 }
156 return is;

222 auto pos = section_and_key.find('.');

224 if (pr.middle.empty())
